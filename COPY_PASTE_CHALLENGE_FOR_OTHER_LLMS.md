```bash
  -filter_complex "
    [0:v]scale=900:900,zoompan=z='zoom+0.002':d=375:s=1080x1920,
    eq=contrast=1.3:brightness=0.05:gamma=1.1,
    colorchannelmixer=rr=1:rb=0.1:gr=0:gg=1:gb=0.1:br=0.1:bg=0.1:bb=1,
    geq='r=r(X,Y)+20*sin(Y*0.1)':'g=g(X,Y)+20*sin(Y*0.1)':'b=b(X,Y)+20*sin(Y*0.1)',
    drawbox=x=0:y=0:w=1080:h=1920:color=black@0.3:t=fill,
    drawtext=text='SHOCK TEXT HERE':fontfile=horror.ttf:fontsize=60:fontcolor=white:x=(w-text_w)/2:y=100,
    [0:v][2:v]overlay=W-270:20[vf]
  " \
  -map "[vf]" -map 1:a \
  -c:v libx264 -preset fast -crf 23 -profile:v high -level 4.2 -pix_fmt yuv420p \
  -c:a aac -b:a 128k -ar 44100 \
  -movflags +faststart \
  -t 15 output.mp4
```

- `lincoln.jpg` = Your Lincoln image
- `audio.mp3` = Your generated voice
- `cashapp_qr.png` = Generated QR code
- `SHOCK TEXT HERE` = Your title text
